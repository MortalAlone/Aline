SQLite 语法

execute()   #执行sql语句
scroll()   #游标滚动
close()   #关闭游标
executemany()   #执行多条sql语句
fetchone()   #从结果中取一条记录
fetchmany()   #从结果中取多条记录
fetchall()   #从结果中取出多条记录

SQLite 基本操作

import sqlite3

conn = sqlite3.connect('test.db')      #如果数据库不存在，那么它就会被创建，最后将返回一个数据库对象

c = conn.cursor()       #创建一个游标对象

c = conn.cursor()

c.execute('CREATE TABLE COMPANY (id integer primary key ,name varchar(20),age integer)')     #创建一个数据表

data = ("1,'leon',22")
c.execute('INSERT INTO COMPANY VALUES (%s)' % (data))   # 插入数据

c.execute('INSERT INTO COMPANY VALUES (?,?,?)',(2,'name2',28))   # 插入数据第二种方法

# 还可以使用executemany()执行多条sql语句，使用executmany()比循环使用excute()执行多条sql语句效率高。
c.executemany('INSERT INTO COMPANY VALUES (?,?,?)',[(3,'name3',19),(4,'name4',26)])

c.execute('SELECT * FROM COMPANY')     #查询数据
res = c.fetchall()  # 取全部数据
res = c.fetchone()  # 取第一条数据
for line in res:
    print("循环fetchall的值>>>",line)


c.execute('UPDATE COMPANY SET name=? WHERE id=? ',('cat',1))    # 修改数据
c.execute('DELETE FROM COMPANY WHERE id=1')                     # 删除数据
print("更新的总行数 :", conn.total_changes)                       # 结果

conn.commit()	#处理事务提交

conn.close()	#关闭游标

data_ToBeInserted = ["170140009,'凡','男',20,'人文'","170140000,'翔','男',20,'GIS'","170140001,'豪','男',20,'GIS'","170140002,'飞','男',20,'人文'","170100001,'亮','男',27,'海棠二楼'"]
